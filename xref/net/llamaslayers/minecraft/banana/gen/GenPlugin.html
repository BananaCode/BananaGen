211 		}
212 
213 		if (radius < 1)
214 			return false;
215 
216 		sender.sendMessage(ChatColor.BLUE + "Starting regeneration...");
217 		final Location start = sender instanceof Player ? ((Player) sender).getLocation()
218 				: new Location(getServer().getWorlds().get(0), 0, 0, 0);
219 		new Runnable() {
220 			private int i = 0;
221 			private int[] coords;
222 			private int taskID = -1;
223 			private final int startX = start.getBlockX() / 16;
224 			private final int startZ = start.getBlockZ() / 16;
225 			private int lastPercent = 0;
226 
227 			@Override
228 			public void run() {
229 				if (i >= coords.length / 2) {
230 					getServer().getScheduler().cancelTask(taskID);
231 					sender.sendMessage(ChatColor.BLUE
232 							+ "Regeneration finished.");
233 					return;
234 				}
235 				start.getWorld().regenerateChunk(startX + coords[i * 2], startZ
236 						+ coords[i * 2 + 1]);
237 				i++;
238 				if (i * 200 / coords.length > lastPercent) {
239 					lastPercent = i * 200 / coords.length;
240 					sender.sendMessage(ChatColor.GREEN + "Regenerating ("
241 							+ lastPercent + "%)...");
242 				}
243 			}
244 
245 			public void schedule() {
246 				coords = new int[8 * radius * radius + 8 * radius + 2];
247 				int _i = 0;
248 				for (int r = 0; r <= radius; r++) {
249 					for (int x = -r; x <= r; x++) {
250 						for (int z = -r; z <= r; z++) {
251 							if (Math.abs(x) == r || Math.abs(z) == r) {
252 								coords[_i++] = x;
253 								coords[_i++] = z;
254 							}
255 						}
256 					}
257 				}
258 
259 				taskID = getServer().getScheduler().scheduleSyncRepeatingTask(GenPlugin.this, this, 1, 10);
260 			}
261 		}.schedule();
262 		return true;
