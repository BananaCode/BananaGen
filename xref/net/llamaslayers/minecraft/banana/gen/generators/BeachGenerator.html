34  	/**
35  	 * @see org.bukkit.generator.ChunkGenerator#generate(org.bukkit.World,
36  	 *      java.util.Random, int, int)
37  	 */
38  	@Override
39  	public byte[] generate(World world, Random random, int chunkX, int chunkZ) {
40  		Map<String, OctaveGenerator> octaves = getWorldOctaves(world);
41  		OctaveGenerator noiseTerrainHeight = octaves.get("terrainHeight");
42  		OctaveGenerator noiseTerrainType = octaves.get("terrainType");
43  		OctaveGenerator noiseTerrainType2 = octaves.get("terrainType2");
44  
45  		chunkX <<= 4;
46  		chunkZ <<= 4;
47  
48  		byte[] b = new byte[272 * 128];
49  
50  		boolean nether = getArg(world, "nether");
51  		byte liquid = nether ? STATIONARY_LAVA : STATIONARY_WATER;
52  		byte sand = nether ? SOUL_SAND : SAND;
53  		byte sandstone = nether ? NETHERRACK : SANDSTONE;
54  		byte stone = nether ? NETHERRACK : STONE;
55  		byte bedrock = BEDROCK;
56  
57  		for (int x = 0; x < 16; x++) {
58  			for (int z = 0; z < 16; z++) {
59  				for (int y = 1; y < 64; y++) {
60  					b[x * 2048 + z * 128 + y] = liquid;
61  				}
62  				int deep = 0;
63  
64  				double _y = noiseTerrainHeight.noise(x + chunkX, z + chunkZ, 0.5, 0.5, true) + 1;
65  				_y = _y * _y * 4;
66  				if (_y < 8) {
67  					_y = _y * _y / 4 - 8;
68  				}
69  
70  				for (int y = 56 + (int) _y; y > 0; y--) {
71  					if (deep < noiseTerrainType.noise(x + chunkX, z + chunkZ, 0.5, 0.5, true) * 3 + 5) {
72  						b[x * 2048 + z * 128 + y] = sand;
73  					} else if (deep < noiseTerrainType2.noise(x + chunkX, z
74  							+ chunkZ, 0.5, 0.5, true) * 4 + 7) {
75  						b[x * 2048 + z * 128 + y] = sandstone;
76  					} else {
77  						b[x * 2048 + z * 128 + y] = stone;
78  					}
79  					deep++;
80  				}
81  				b[x * 2048 + z * 128] = bedrock;
82  			}
83  		}
84  
85  		return b;
86  	}
