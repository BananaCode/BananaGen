239 	@Override
240 	public boolean canSpawn(World world, int x, int z) {
241 		Block block = world.getHighestBlockAt(x, z).getRelative(BlockFace.DOWN);
242 		return !block.isLiquid() && !block.isEmpty()
243 				&& !FORBIDDEN_SPAWN_FLOORS.contains(block.getType());
