23  public class GlowstonePopulator extends BananaBlockPopulator {
24  	private static final BlockFace[] faces = {BlockFace.DOWN, BlockFace.EAST,
25  		BlockFace.NORTH, BlockFace.SOUTH, BlockFace.UP, BlockFace.WEST};

27  	/**
28  	 * Populates a world with glowstone. Easily configurable (but results in
29  	 * more rare glowstone) by modifying the suitable()-method.
30  	 */
31  	@Override
32  	public void populate(World world, Random random, Chunk source) {
33  		for (int i = 0; i < 2; i++) {
34  			int x = random.nextInt(16);
35  			int y = random.nextInt(128);
36  			int z = random.nextInt(16);
37  			while (!suitable(y)) {
38  				y = random.nextInt(128);
39  			}
40  			Block block = source.getBlock(x, y, z);
41  			// Only populates if the "target" location is air & there is netherrack
42  			// above it. (and only checked in Y 114-127 or 52-72 for efficiency,
43  			// might be changed later)
44  			if (block.getTypeId() != AIR) {
45  				return;
46  			}
47  			if (block.getRelative(BlockFace.UP).getTypeId() != NETHERRACK) {
48  				return;
49  			}
50  			block.setTypeId(GLOWSTONE);
51  
52  			for (int j = 0; j < 1500; j++) {
53  				Block current = block.getRelative(random.nextInt(8) - random.nextInt(8),
54  												  random.nextInt(12),
55  												  random.nextInt(8) - random.nextInt(8));
56  				if (current.getTypeId() != AIR) {
57  					continue;
58  				}
59  				int count = 0;
60  				for (BlockFace face : faces) {
61  					if (current.getRelative(face).getTypeId() == GLOWSTONE) {
62  						count++;
63  					}
64  				}
65  
66  				if (count == 1) {
67  					current.setTypeId(GLOWSTONE);
68  				}
69  			}
70  		}
71  	}

73  	/**
74  	 * Checks if the given Y-coordinate is "suitable" for glowstone generation. <br />
75  	 * <br />
76  	 * If the Y-coordinate is 52-72, it is OK. <br />
77  	 * If the Y-coordinate is 114-127, it is OK.
78  	 *
79  	 * @param y
80  	 *            Y-coordinate
81  	 * @return true if the given Y-coordinate is suitable for glowstone
82  	 *         generation
83  	 */
84  	private static boolean suitable(int y) {
85  		if (y > 113 && y < 128) {
86  			return true;
87  		}
88  		if (y > 51 && y < 73) {
89  			return true;
90  		}
91  		return false;
92  	}
