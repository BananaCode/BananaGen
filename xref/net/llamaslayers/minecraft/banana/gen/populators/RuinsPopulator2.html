27  	@Override
28  	public void populate(World world, Random random, Chunk source) {
29  		if (getArg(world, "noruins2"))
30  			return;
31  
32  		byte material;
33  		try {
34  			material = (byte) Material.valueOf(getArgString(world, "ruins2material", "COBBLESTONE")).getId();
35  		} catch (IllegalArgumentException ex) {
36  			material = COBBLESTONE;
37  			// TODO: complain?
38  		}
39  
40  		if (random.nextInt(100) < RUINS_CHANCE) {
41  			int count = random.nextInt(MAX_CLUMP_SIZE - MIN_CLUMP_SIZE + 1) + MIN_CLUMP_SIZE;
42  			int startX = random.nextInt(16) + source.getX() * 16;
43  			int startZ = random.nextInt(16) + source.getZ() * 16;
44  
45  			Block[] seeds = new Block[count];
46  			for (int i = 0; i < count; i++) {
47  				int x = startX + random.nextInt(5) - random.nextInt(5);
48  				int z = startZ + random.nextInt(5) - random.nextInt(5);
49  				seeds[i] = world.getBlockAt(x, world.getHighestBlockYAt(x, z), z);
50  			}
51  			for (Block seed : seeds)
52  				buildRuin(seed, random, material);
53  		}
54  	}
55  
56  	private static void buildRuin(Block start, Random random, byte material) {
57  		if (start.isEmpty())
58  			start = start.getRelative(BlockFace.DOWN);
59  		int radius = random.nextInt(3) + 1;
60  		int r2 = radius * radius;
61  		int height = random.nextInt(5) + 3;
62  		for (int x = -radius; x <= radius; x++) {
63  			for (int z = -radius; z <= radius; z++) {
64  				for (int y = height * (r2 - (x * x + z * z)) / r2 + random.nextInt(3); y >= 0; y--) {
65  					start.getRelative(x, y, z).setTypeId(material);
66  				}
67  			}
68  		}
69  	}
