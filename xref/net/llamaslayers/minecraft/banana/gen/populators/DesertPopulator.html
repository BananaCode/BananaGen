15   * BlockPopulator that turns deserts into sand and places cacti.

17   * @author codename_B
18   */
19  public class DesertPopulator extends BananaBlockPopulator {
20  	/**
21  	 * @see org.bukkit.generator.BlockPopulator#populate(org.bukkit.World,
22  	 *      java.util.Random, org.bukkit.Chunk)
23  	 */
24  	@Override
25  	public void populate(World world, Random random, Chunk chunk) {
26  		if (getArg(world, "nodeserts"))
27  			return;
28  		Material matSand = getArg(world, "nether") ? Material.SOUL_SAND
29  				: Material.SAND;
30  		Material matDirt = getArg(world, "nether") ? Material.NETHERRACK
31  				: Material.DIRT;
32  
33  		for (int x = 0; x < 16; x++) {
34  			for (int z = 0; z < 16; z++) {
35  				int tx = (chunk.getX() << 4) + x;
36  				int tz = (chunk.getZ() << 4) + z;
37  				int y = world.getHighestBlockYAt(tx, tz);
38  
39  				Block block = chunk.getBlock(x, y, z).getRelative(BlockFace.DOWN);
40  				if (block.getBiome() != Biome.DESERT) {
41  					continue;
42  				}
43  
44  				// Set top few layers of grass/dirt to sand
45  				for (int i = 0; i < 5; ++i) {
46  					Block b2 = block.getRelative(0, -i, 0);
47  					if (b2.getType() == Material.GRASS
48  							|| b2.getType() == matDirt) {
49  						b2.setType(matSand);
50  					}
51  				}
52  
53  				// Generate cactus
54  				if (block.getType() == matSand) {
55  					if (random.nextInt(20) == 0) {
56  						// Make sure it's surrounded by air
57  						Block base = block.getRelative(BlockFace.UP);
58  						if (base.getTypeId() == 0
59  								&& base.getRelative(BlockFace.NORTH).getTypeId() == 0
60  								&& base.getRelative(BlockFace.EAST).getTypeId() == 0
61  								&& base.getRelative(BlockFace.SOUTH).getTypeId() == 0
62  								&& base.getRelative(BlockFace.WEST).getTypeId() == 0) {
63  							generateCactus(base, random.nextInt(4));
64  						}
65  					}
66  				}
67  			}
68  		}
69  	}
70  
71  	private void generateCactus(Block block, int height) {
72  		if (getArg(block.getWorld(), "nether")) {
73  			block.setType(Material.FIRE);
74  		} else {
75  			for (int i = 0; i < height; ++i) {
76  				block.getRelative(0, i, 0).setType(Material.CACTUS);
77  			}
78  		}
79  	}
