18  public class RuinsPopulator extends BananaBlockPopulator {
19  	private static final int MAX_RUINS = 3;
20  	private static final int RUINS_CHANCE = 20;
21  	private static final Material RUINS_MATERIAL = Material.COBBLESTONE;
22  	private static final Material NETHER_RUINS_MATERIAL = Material.GLOWSTONE;
23  	private static final BlockFace[] directions = new BlockFace[] {
24  			BlockFace.NORTH, BlockFace.SOUTH, BlockFace.EAST, BlockFace.WEST };
25  
26  	/**
27  	 * @see org.bukkit.generator.BlockPopulator#populate(org.bukkit.World,
28  	 *      java.util.Random, org.bukkit.Chunk)
29  	 */
30  	@Override
31  	public void populate(World world, Random random, Chunk source) {
32  		if (getArg(world, "noruins"))
33  			return;
34  		ChunkSnapshot snapshot = source.getChunkSnapshot();
35  		int ruins = 0;
36  		while (random.nextInt(100) < RUINS_CHANCE
37  				&& ruins < MAX_RUINS) {
38  			int startX = random.nextInt(14) + 1;
39  			int startZ = random.nextInt(14) + 1;
40  			int startY = snapshot.getHighestBlockYAt(startX, startZ);
41  			int startHeight = random.nextInt(5) + 1;
42  
43  			BlockFace direction1 = directions[random.nextInt(directions.length)];
44  			BlockFace direction2 = directions[random.nextInt(directions.length)];
45  
46  			if (source.getBlock(startX, startY - 1, startZ).isLiquid()) {
47  				ruins++;
48  				continue;
49  			}
50  
51  			int height = startHeight;
52  			int x = startX;
53  			int z = startZ;
54  			while (height > 0 && 0 <= x && x < 16 && 0 <= z && z < 16) {
55  				int y = snapshot.getHighestBlockYAt(x, z);
56  				while (source.getBlock(x, y - 1, z).isLiquid() && y > 0) {
57  					y--;
58  				}
59  				for (; y < startY + height; y++) {
60  					source.getBlock(x, y, z).setType(getArg(world, "nether") ? NETHER_RUINS_MATERIAL
61  							: RUINS_MATERIAL);
62  				}
63  
64  				height -= random.nextInt(3);
65  
66  				x += direction1.getModX();
67  				z += direction1.getModZ();
68  			}
69  
70  			if (direction1 != direction2) {
71  				height = startHeight;
72  				x = startX;
73  				z = startZ;
74  				while (height > 0 && 0 <= x && x < 16 && 0 <= z && z < 16) {
75  					int y = snapshot.getHighestBlockYAt(x, z);
76  					while (source.getBlock(x, y - 1, z).isLiquid() && y > 0) {
77  						y--;
78  					}
79  					for (; y < startY + height; y++) {
80  						source.getBlock(x, y, z).setType(getArg(world, "nether") ? NETHER_RUINS_MATERIAL
81  								: RUINS_MATERIAL);
82  					}
83  
84  					height -= random.nextInt(3);
85  
86  					x += direction2.getModX();
87  					z += direction2.getModZ();
88  				}
89  			}
90  
91  			ruins++;
92  		}
93  	}
