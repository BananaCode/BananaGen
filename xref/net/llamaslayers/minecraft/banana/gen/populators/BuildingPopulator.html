57  	BuildingPopulator(String category, String buildingType) {
58  		buildings = new HashMap<String, Building>();
59  		groups = new HashMap<String, List<Building>>();
60  		Enumeration<JarEntry> schematics;
61  		JarFile jar;
62  		try {
63  			URLClassLoader classLoader = (URLClassLoader) getClass().getClassLoader();
64  			URL jarUrl = classLoader.getURLs()[0];
65  			jar = new JarFile(jarUrl.toString().replace("file:/", File.separator).replace("%20", " "));
66  			schematics = jar.entries();
67  		} catch (IOException ex) {
68  			ex.printStackTrace();
69  			return;
70  		}
71  		while (schematics.hasMoreElements()) {
72  			JarEntry schematic = schematics.nextElement();
73  			if (!schematic.getName().startsWith(buildingType + "/" + category + "/")) {
74  				continue;
75  			}
76  
77  			String buildingName = schematic.getName();
78  			try {
79  				buildingName = buildingName.substring(buildingName.lastIndexOf('/') + 1);
80  				buildingName = buildingName.substring(0, buildingName.lastIndexOf('.'));
81  			} catch (StringIndexOutOfBoundsException ex) {
82  				continue;
83  			}
84  			Building building = new Building(buildingName);
85  			buildings.put(buildingName, building);
86  			try {
87  				if (buildingType.equals("buildings")) {
88  					BufferedReader in = new BufferedReader(new InputStreamReader(
89  							jar.getInputStream(schematic)));
90  					while (building.parseLine(in.readLine()));
91  					in.close();
92  					building.produceRotatedVersions();
93  				} else if (buildingType.equals("schematics")) {
94  					NBTInputStream in = new NBTInputStream(new GZIPInputStream(
95  							jar.getInputStream(schematic)));
96  
97  					try {
98  						building.parseSchematic(in);
99  					} catch (NoSuchFieldException ex) {
100 						throw new IOException(ex);
101 					} finally {
102 						in.close();
103 					}
104 				} else {
105 					throw new IllegalStateException("BuildingPopulator(String, String) should not be used directly");
106 				}
107 			} catch (IOException ex) {
108 				ex.printStackTrace();
109 				buildings.remove(buildingName);
110 			}
111 		}
112 	}

115 	 * Get child tag of a NBT structure. From WorldEdit.
116 	 *
117 	 * @param items
118 	 * @param key
119 	 * @param expected
120 	 * @return child tag
121 	 * @throws DataException
122 	 */
123 	static Tag getChildTag(Map<String, Tag> items, String key,
124 						   Class<? extends Tag> expected) throws NoSuchFieldException {
125 
126 		if (!items.containsKey(key)) {
127 			throw new NoSuchFieldException("Schematic file is missing a \"" + key + "\" tag");
128 		}
129 		Tag tag = items.get(key);
130 		if (!expected.isInstance(tag)) {
131 			throw new NoSuchFieldException(
132 					key + " tag is not of tag type " + expected.getName());
133 		}
134 		return tag;
135 	}

280 		// Logic derived heavily from https://github.com/sk89q/worldedit/blob/master/src/main/java/com/sk89q/worldedit/CuboidClipboard.java
281 		void parseSchematic(NBTInputStream in) throws NoSuchFieldException, IOException {
282 			Vector origin = new Vector();
283 			Vector offset = new Vector();
284 
285 			// Schematic tag
286 			CompoundTag schematicTag = (CompoundTag) in.readTag();
287 			if (!schematicTag.getName().equals("Schematic")) {
288 				throw new IOException("Tag \"Schematic\" does not exist or is not first");
289 			}
290 
291 			// Check
292 			Map<String, Tag> schematic = schematicTag.getValue();
293 			if (!schematic.containsKey("Blocks")) {
294 				throw new NoSuchFieldException("Schematic file is missing a \"Blocks\" tag");
295 			}
296 
297 			// Get information
298 			short width = (Short) getChildTag(schematic, "Width", ShortTag.class).getValue();
299 			short length = (Short) getChildTag(schematic, "Length", ShortTag.class).getValue();
300 			short height = (Short) getChildTag(schematic, "Height", ShortTag.class).getValue();
301 
302 			try {
303 				int originX = (Integer) getChildTag(schematic, "WEOriginX", IntTag.class).getValue();
304 				int originY = (Integer) getChildTag(schematic, "WEOriginY", IntTag.class).getValue();
305 				int originZ = (Integer) getChildTag(schematic, "WEOriginZ", IntTag.class).getValue();
306 				origin = new Vector(originX, originY, originZ);
307 			} catch (NoSuchFieldException e) {
308 				// No origin data
309 			}
310 
311 			try {
312 				int offsetX = (Integer) getChildTag(schematic, "WEOffsetX", IntTag.class).getValue();
313 				int offsetY = (Integer) getChildTag(schematic, "WEOffsetY", IntTag.class).getValue();
314 				int offsetZ = (Integer) getChildTag(schematic, "WEOffsetZ", IntTag.class).getValue();
315 				offset = new Vector(offsetX, offsetY, offsetZ);
316 			} catch (NoSuchFieldException e) {
317 				// No offset data
318 			}
319 
320 			// Check type of Schematic
321 			String materials = (String) getChildTag(schematic, "Materials", StringTag.class).getValue();
322 			if (!materials.equals("Alpha")) {
323 				throw new IOException("Schematic file is not an Alpha schematic");
324 			}
325 
326 			// Get blocks
327 			byte[] _blocks = (byte[]) getChildTag(schematic, "Blocks", ByteArrayTag.class).getValue();
328 			byte[] _blockData = (byte[]) getChildTag(schematic, "Data", ByteArrayTag.class).getValue();
329 
330 			// Tile entities are not implemented
331 
332 			// begin implementation-specific section
333 			spawnDark = true;
334 			spawnLight = true;
335 			for (int x = 0; x < width; x++) {
336 				for (int y = 0; y < height; y++) {
337 					for (int z = 0; z < length; z++) {
338 						BuildingBlock block = new BuildingBlock(
339 								x + origin.getBlockX() + offset.getBlockX(),
340 								y + origin.getBlockY() + offset.getBlockY(),
341 								z + origin.getBlockZ() + offset.getBlockZ(),
342 								_blocks[(x * length + z) * height + y],
343 								_blockData[(x * length + z) * height + y]);
344 						maxX = Math.max(maxX, block.x);
345 						minX = Math.min(minX, block.x);
346 						maxY = Math.max(maxY, block.y);
347 						minY = Math.min(minY, block.y);
348 						maxZ = Math.max(maxZ, block.z);
349 						minZ = Math.min(minZ, block.z);
350 						blocks.add(block);
351 					}
352 				}
353 			}
354 			// end implementation-specific section
355 		}

357 		boolean parseLine(String line) {
358 			if (line == null) {
359 				return false;
360 			}
361 			if (line.trim().length() == 0) {
362 				return true;
363 			}
364 
365 			if (line.equals("[META]")) {
366 				section = META;
367 				return true;
368 			}
369 			if (line.equals("[DATA]")) {
370 				section = DATA;
371 				return true;
372 			}
373 			if (section == META) {
374 				String[] parts = line.split("=", 2);
375 				String key = parts[0].toLowerCase();
376 				String value = parts[1].toLowerCase();
377 
378 				if (key.equals("version")) {
379 					if (!value.equals("2.0")) {
380 						throw new RuntimeException("In " + name
381 								+ ".bo2: Expecting version 2.0; got version "
382 								+ value);
383 					}
384 					return true;
385 				}
386 				if (key.equals("spawninbiome")) {
387 					if (biomes != null) {
388 						throw new RuntimeException("In " + name
389 								+ ".bo2: multiple spawninbiome declarations");
390 					}
391 					if (value.equals("all")) {
392 						biomes = Biome.values();
393 					} else {
394 						String[] biomeNames = value.toUpperCase().split(",");
395 						biomes = new Biome[biomeNames.length];
396 						for (int i = 0; i < biomeNames.length; i++) {
397 							biomes[i] = Biome.valueOf(biomeNames[i].replace(' ', '_').replace("RAIN_FOREST", "RAINFOREST"));
398 						}
399 					}
400 					Arrays.sort(biomes);
401 					return true;
402 				}
403 				if (key.equals("randomrotation")) {
404 					randomRotation = Boolean.parseBoolean(value);
405 					return true;
406 				}
407 				if (key.equals("spawnonblocktype")) {
408 					String[] blockTypes = value.split(",");
409 					spawnBlockType = new int[blockTypes.length];
410 					for (int i = 0; i < blockTypes.length; i++) {
411 						spawnBlockType[i] = Integer.parseInt(blockTypes[i]);
412 					}
413 					Arrays.sort(spawnBlockType);
414 					return true;
415 				}
416 				if (key.equals("spawnsunlight")) {
417 					spawnLight = Boolean.parseBoolean(value);
418 					return true;
419 				}
420 				if (key.equals("spawndarkness")) {
421 					spawnDark = Boolean.parseBoolean(value);
422 					return true;
423 				}
424 				if (key.equals("spawnwater")) {
425 					spawnWater = Boolean.parseBoolean(value);
426 					return true;
427 				}
428 				if (key.equals("spawnlava")) {
429 					spawnLava = Boolean.parseBoolean(value);
430 					return true;
431 				}
432 				if (key.equals("underfill")) {
433 					underfill = Boolean.parseBoolean(value);
434 					return true;
435 				}
436 				if (key.equals("dig")) {
437 					dig = Boolean.parseBoolean(value);
438 					return true;
439 				}
440 				if (key.equals("tree")) {
441 					return true; // TODO, currently unsupported
442 				}
443 				if (key.equals("branch")) {
444 					return true; // TODO, currently unsupported
445 				}
446 				if (key.equals("diggingbranch")) {
447 					return true; // TODO, currently unsupported
448 				}
449 				if (key.equals("needsfoundation")) {
450 					needsFoundation = Boolean.parseBoolean(value);
451 					return true;
452 				}
453 				if (key.equals("rarity")) {
454 					rarity = Integer.parseInt(value);
455 					return true;
456 				}
457 				if (key.equals("collisionpercentage")) {
458 					collisionPercent = Integer.parseInt(value);
459 					return true;
460 				}
461 				if (key.equals("spawnelevationmin")) {
462 					heightMin = Integer.parseInt(value);
463 					return true;
464 				}
465 				if (key.equals("spawnelevationmax")) {
466 					heightMax = Integer.parseInt(value);
467 					return true;
468 				}
469 				if (key.equals("groupid")) {
470 					group = value.toLowerCase();
471 					addToGroup(group, this);
472 					return true;
473 				}
474 				if (key.equals("groupfrequencymin")
475 						|| key.equals("groupfrequencymax")
476 						|| key.equals("groupseperationmin")
477 						|| key.equals("groupseperationmax")) {
478 					return true; // TODO, currently unsupported
479 				}
480 				if (key.equals("branchlimit")) {
481 					return true; // TODO, currently unsupported
482 				}
483 			}
484 			if (section == DATA) {
485 				try {
486 					BuildingBlock block = new BuildingBlock(line);
487 					maxX = Math.max(maxX, block.x);
488 					minX = Math.min(minX, block.x);
489 					maxY = Math.max(maxY, block.y);
490 					minY = Math.min(minY, block.y);
491 					maxZ = Math.max(maxZ, block.z);
492 					minZ = Math.min(minZ, block.z);
493 					blocks.add(block);
494 					return true;
495 				} catch (OperationNotSupportedException ex) {
496 					throw new RuntimeException("In " + name + ".bo2: "
497 							+ ex.getMessage());
498 				}
499 			}
500 
501 			throw new RuntimeException("Unparsed line in " + name + ".bo2: "
502 					+ line);
503 		}

514 		/**
515 		 * Determine if this building can be placed on a given block
516 		 *
517 		 * @param block
518 		 *            The block where location 0, 0, 0 would be
519 		 * @return true if the building can be placed, false if it cannot
520 		 */
521 		public boolean canPlaceOn(Block block) {
522 			if (block.getY() > heightMax || block.getY() < heightMin) //debug("Start Y out of range");
523 			{
524 				return false;
525 			}
526 			if (block.getY() + maxY > 127 || block.getY() + minY < 0) //debug("End Y out of range");
527 			{
528 				return false;
529 			}
530 
531 			if (spawnBlockType != null
532 					&& Arrays.binarySearch(spawnBlockType, block.getRelative(BlockFace.DOWN).getTypeId()) < 0) //debug("Trying to spawn on ", Integer.toString(block.getTypeId()),
533 			//		", allowed are ", Arrays.toString(spawnBlockType));
534 			{
535 				return false;
536 			}
537 			if (biomes != null
538 					&& Arrays.binarySearch(biomes, block.getBiome()) < 0) //debug("Trying to spawn in ", block.getBiome().name(), ", allowed are ", Arrays.toString(biomes));
539 			{
540 				return false;
541 			}
542 
543 			byte light = block.getLightLevel();
544 			if (block.getWorld().getHighestBlockYAt(block.getX(), block.getZ()) - 1 == block.getY()) {
545 				light = 15;
546 			}
547 			if (light < 4 && !spawnDark) //debug("Trying to spawn in the dark, but not allowed to");
548 			{
549 				return false;
550 			}
551 			if (light >= 4 && !spawnLight) //debug("Trying to spawn in the light, but not allowed to");
552 			{
553 				return false;
554 			}
555 
556 			if (!spawnWater
557 					&& (checkFor(Material.WATER, block) || checkFor(Material.STATIONARY_WATER, block))) //debug("Water in the way");
558 			{
559 				return false;
560 			}
561 			if (!spawnLava
562 					&& (checkFor(Material.LAVA, block) || checkFor(Material.STATIONARY_LAVA, block))) //debug("Lava in the way");
563 			{
564 				return false;
565 			}
566 
567 			if (getCollisionPercent(block) > collisionPercent) //debug("Collision: ", Integer.toString(getCollisionPercent(block)), " > ", Integer.toString(collisionPercent));
568 			{
569 				return false;
570 			}
571 
572 			if (needsFoundation && !underfill) {
573 				if (!isSupported(block)) //debug("Unsupported, but needs foundation");
574 				{
575 					return false;
576 				}
577 			}
578 
579 			if (checkFor(Material.BEDROCK, block)) //debug("Trying to spawn on bedrock");
580 			{
581 				return false;
582 			}
583 			//debug("Success!");
584 			return true;
585 		}

594 		private boolean isSupported(Block block) {
595 			for (BuildingBlock b : blocks) {
596 				Block relative = block.getRelative(b.x, b.y, b.z);
597 
598 				if (b.y == minY) {
599 					relative = relative.getRelative(BlockFace.DOWN);
600 					if (!relative.isLiquid() && !relative.isEmpty()) {
601 						return false;
602 					}
603 				}
604 			}
605 			return true;
606 		}

608 		private int getCollisionPercent(Block block) {
609 			int count = 0;
610 			for (BuildingBlock b : blocks) {
611 				Block relative = block.getRelative(b.x, b.y, b.z);
612 				if (!relative.isLiquid() && !relative.isEmpty()) {
613 					count++;
614 				}
615 			}
616 			return count * 100 / blocks.size();
617 		}

619 		private boolean checkFor(Material material, Block block) {
620 			for (int x = minX; x <= maxX; x++) {
621 				for (int y = minY; y <= maxY; y++) {
622 					for (int z = minZ; z <= maxZ; z++) {
623 						if (block.getRelative(x, y, z).getType() == material) {
624 							return true;
625 						}
626 					}
627 				}
628 			}
629 			return false;
630 		}

763 		BuildingBlock(String line) throws OperationNotSupportedException {
764 			String[] halves = line.split(":");
765 			String[] coords = halves[0].split(",");
766 			x = Integer.parseInt(coords[0]);

769

771 			remaining.find();
772 			id = (byte) Integer.parseInt(remaining.group(1));
773 			data = remaining.group(2) == null ? 0

777 			}
