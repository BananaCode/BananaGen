32  		if (random.nextInt(100) < 1) {
33  			int sizeX = 16;
34  			int sizeY = 12;

39  			Material matDecor = Material.WEB;

41  			//Editing is over
42  			int lengthH = sizeX / 2;
43  			int heightH = sizeY / 2;
44  			int widthH = sizeZ / 2;
45  			int centerX = source.getX() * 16 + 8;
46  			int centerY = world.getHighestBlockYAt(source.getX() * 16 + 8, source.getZ() * 16 + 8) / 2;
47  			int centerZ = source.getZ() * 16 + 8;
48  			int minX = centerX - lengthH;
49  			int maxX = centerX + lengthH;
50  			int minY = centerY - heightH;
51  			int maxY = centerY + heightH;
52  			int minZ = centerZ - widthH;
53  			int maxZ = centerZ + widthH;
54  
55  			// Step 1: Cuboid generation around the entire area
56  			for (int x = minX; x <= maxX; x++) {
57  				for (int y = minY; y <= maxY; y++) {
58  					world.getBlockAt(x, y, minZ).setType(matWalls);
59  					world.getBlockAt(x, y, maxZ).setType(matWalls);
60  				}
61  			}
62  
63  			for (int y = minY; y <= maxY; y++) {
64  				for (int z = minZ; z <= maxZ; z++) {
65  					world.getBlockAt(minX, y, z).setType(matWalls);
66  					world.getBlockAt(maxX, y, z).setType(matWalls);
67  				}
68  			}
69  
70  			for (int z = minZ; z <= maxZ; z++) {
71  				for (int x = minX; x <= maxX; x++) {
72  					world.getBlockAt(x, minY, z).setType(matWalls);
73  					world.getBlockAt(x, maxY, z).setType(matWalls);
74  				}
75  			}
76  
77  			// Step 2: Add netherrack and web noise to walls (XY, YZ)
78  			minX++;
79  			maxX--;
80  			minY++;
81  			maxY--;
82  			minZ++;
83  			maxZ--;
84  
85  			for (int x = minX; x <= maxX; x++) {
86  				for (int y = minY; y <= maxY; y++) {
87  					world.getBlockAt(x, y, minZ).setType(pickDecor(random, matDecor, matWalls));
88  					world.getBlockAt(x, y, maxZ).setType(pickDecor(random, matDecor, matWalls));
89  				}
90  			}
91  
92  			for (int y = minY; y <= maxY; y++) {
93  				for (int z = minZ; z <= maxZ; z++) {

95  					world.getBlockAt(maxX, y, z).setType(pickDecor(random, matDecor, matWalls));
96  				}
97  			}
98  
99  			// Step 3: Generate a floor of soul sand/mobs and a ceiling of netherrack
100 			for (int z = minZ; z <= maxZ; z++) {
101 				for (int x = minX; x <= maxX; x++) {
102 					int floor = random.nextInt(spawnerChance); // spawner rate
103 					Block block = world.getBlockAt(x, minY, z);
104 					if (floor < 12) {
105 						block.setType(Material.MOB_SPAWNER);
106 						BlockState state = block.getState();
107 						if (state instanceof CreatureSpawner) {
108 							CreatureSpawner spawner = (CreatureSpawner) state;
109 							if (floor <= 2) {
110 								spawner.setCreatureType(CreatureType.ZOMBIE);
111 							} else if (floor >= 3 && floor <= 7) {
112 								spawner.setCreatureType(CreatureType.SPIDER);
113 							} else if (floor >= 8 && floor <= 10) {
114 								spawner.setCreatureType(CreatureType.SKELETON);
115 							} else {
116 								spawner.setCreatureType(CreatureType.GHAST);
117 							}
118 						} else {
119 							Bukkit.getServer().getLogger().severe("Spawner is not giving correct state, returned "
120 									+ state
121 									+ " (block type: "
122 									+ block.getType()
123 									+ ", state type: " + state.getType() + ")");
124 						}
125 					} else {
126 						block.setType(matFloor);
127 					}
128 					world.getBlockAt(x, maxY, z).setType(matWalls);
129 				}
130 			}
131 
132 			// Step 4: Stalagmites
133 			minX++;
134 			maxX--;
135 			minY++;
136 			maxY--;
137 			minZ++;
138 			maxZ--;
139 
140 			for (int z = minZ; z <= maxZ; z++) {
141 				for (int x = minX; x <= maxX; x++) {
142 					for (int y = minY; y <= maxY; y++) {
143 						Block block = world.getBlockAt(x, y, z);
144 						if (block.getRelative(BlockFace.DOWN).getType() != Material.AIR
145 								&& block.getRelative(BlockFace.DOWN).getType() != matDecor) {
146 							int rand = random.nextInt(10);
147 							if (rand <= 6) {
148 								world.getBlockAt(x, y, z).setType(matFloor);
149 							} else {
150 								world.getBlockAt(x, y, z).setType(pickDecor(random, matDecor, Material.AIR));
151 							}
152 						} else {
153 							world.getBlockAt(x, y, z).setType(Material.AIR);
154 						}
155 					}
156 				}
157 			}
158 		}
