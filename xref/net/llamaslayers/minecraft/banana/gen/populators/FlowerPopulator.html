19  public class FlowerPopulator extends BananaBlockPopulator {
20  	/**
21  	 * @see org.bukkit.generator.BlockPopulator#populate(org.bukkit.World,
22  	 *      java.util.Random, org.bukkit.Chunk)
23  	 */
24  	@Override
25  	public void populate(World world, Random random, Chunk source) {
26  		if (getArg(world, "noflowers"))
27  			return;
28  		if (getArg(world, "nether"))
29  			return;
30  
31  		for (int x = 0; x < 16; x++) {
32  			for (int z = 0; z < 16; z++) {
33  				int cx = (source.getX() << 4) + x;
34  				int cz = (source.getZ() << 4) + z;
35  				int y = world.getHighestBlockYAt(cx, cz);
36  
37  				Block block = source.getBlock(x, y, z);
38  				if (block.getType() == Material.AIR
39  						&& block.getRelative(BlockFace.DOWN).getType() == Material.GRASS) {
40  					if (block.getBiome() == Biome.PLAINS) {
41  						int n = random.nextInt(64);
42  						if (n < 1) {
43  							block.setType(Material.RED_ROSE);
44  						} else if (n < 4) {
45  							block.setType(Material.YELLOW_FLOWER);
46  						}
47  					} else if (block.getBiome() == Biome.SHRUBLAND
48  							|| block.getBiome() == Biome.SAVANNA) {
49  						int n = random.nextInt(256);
50  						if (n < 2) {
51  							block.setType(Material.RED_ROSE);
52  						} else if (n < 3) {
53  							block.setType(Material.YELLOW_FLOWER);
54  						} else if (n < 16) {
55  							block.setType(Material.LONG_GRASS);
56  							block.setData((byte) 1);
57  						}
58  					} else if (block.getBiome() == Biome.FOREST
59  							|| block.getBiome() == Biome.SEASONAL_FOREST) {
60  						int n = random.nextInt(256);
61  						if (n < 16) {
62  							block.setType(Material.LONG_GRASS);
63  							block.setData((byte) 2);
64  						}
65  					}
66  				}
67  			}
68  		}
69  	}
