27  	@Override
28  	public void populate(World world, Random random, Chunk source) {
29  		SimplexNoiseGenerator noise = new SimplexNoiseGenerator(world);
30  		ChunkSnapshot snapshot = source.getChunkSnapshot();
31  
32  		// Randomly turn exposed stone to treasure
33  		for (int x = 0; x < 16; x++) {
34  			for (int z = 0; z < 16; z++) {
35  				int y = snapshot.getHighestBlockYAt(x, z);
36  				Block block = source.getBlock(x, y - 1, z);
37  
38  				if (block.getType() == Material.STONE
39  						&& random.nextInt(1024) == 0) {
40  					placeChest(random, block);
41  				}
42  			}
43  		}
44  
45  		// Go go dungeons
46  		double density = noise.noise(source.getX(), source.getZ());
47  		if (density > 0.8) {
48  			int roomCount = (int) (density * 10) - 3;
49  
50  			for (int i = 0; i < roomCount; i++) {
51  				if (random.nextBoolean()) {
52  					int x = (source.getX() << 4) + random.nextInt(16);
53  					int z = (source.getZ() << 4) + random.nextInt(16);
54  					int y = 12 + random.nextInt(22);
55  
56  					int sizeX = random.nextInt(12) + 5;
57  					int sizeY = random.nextInt(6) + 4;
58  					int sizeZ = random.nextInt(12) + 5;
59  
60  					generateRoom(x, y, z, sizeX, sizeY, sizeZ, world, random);
61  				}
62  			}
63  		}
64  	}

66  	private static void generateRoom(int posX, int posY, int posZ, int sizeX,
67  									 int sizeY, int sizeZ, World world, Random random) {
68  		// Fill with air
69  		for (int x = posX; x < posX + sizeX; x++) {
70  			for (int y = posY; y < posY + sizeY; y++) {
71  				for (int z = posZ; z < posZ + sizeZ; z++) {
72  					placeBlock(world, x, y, z, Material.AIR);
73  				}
74  			}
75  		}
76  
77  		// Spawners
78  		int numSpawners = 1 + random.nextInt(2);
79  		for (int i = 0; i < numSpawners; ++i) {
80  			int x = posX + random.nextInt(sizeX);
81  			int z = posZ + random.nextInt(sizeZ);
82  			placeSpawner(random, world.getBlockAt(x, posY, z));
83  		}
84  
85  		// Chests
86  		int numChests = numSpawners + random.nextInt(2);
87  		for (int i = 0; i < numChests; ++i) {
88  			int x = posX + random.nextInt(sizeX);
89  			int z = posZ + random.nextInt(sizeZ);
90  			placeChest(random, world.getBlockAt(x, posY, z));
91  		}
92  
93  		for (int x = posX - 1; x <= posX + sizeX; x++) {
94  			for (int z = posZ - 1; z <= posZ + sizeZ; z++) {
95  				placeBlock(world, x, posY - 1, z, pickStone(random));
96  				placeBlock(world, x, posY + sizeY, z, pickStone(random));
97  			}
98  		}
99  
100 		for (int y = posY - 1; y <= posY + sizeX; y++) {
101 			for (int z = posZ - 1; z <= posZ + sizeZ; z++) {
102 				placeBlock(world, posX - 1, y, z, pickStone(random));
103 				placeBlock(world, posX + sizeX, y, z, pickStone(random));
104 			}
105 		}
106 
107 		for (int x = posX - 1; x <= posX + sizeX; x++) {
108 			for (int y = posY - 1; y <= posY + sizeY; y++) {
109 				placeBlock(world, x, y, posZ - 1, pickStone(random));
110 				placeBlock(world, x, y, posZ + sizeZ, pickStone(random));
111 			}
112 		}
113 	}

120 	private static void placeSpawner(Random random, Block block) {
121 		CreatureType[] types = new CreatureType[] {
122 			CreatureType.SKELETON, CreatureType.ZOMBIE,
123 			CreatureType.CREEPER, CreatureType.SPIDER
124 		};
125 
126 		block.setType(Material.MOB_SPAWNER);
127 		BlockState state = block.getState();
128 		if (state instanceof CreatureSpawner) {
129 			((CreatureSpawner) state).setCreatureType(types[random.nextInt(types.length)]);
130 		} else {
131 			Bukkit.getServer().getLogger().severe("Spawner is not giving correct state, returned "
132 					+ state
133 					+ " (block type: "
134 					+ block.getType()
135 					+ ", state type: " + state.getType() + ")");
136 		}
137 	}

139 	private static void placeChest(Random random, Block block) {
140 		block.setType(Material.CHEST);
141 		BlockState state = block.getState();
142 		if (state instanceof Chest) {
143 			Inventory chest = ((Chest) state).getInventory();
144 
145 			for (int i = 0; i < 5; i++) {
146 				chest.setItem(random.nextInt(chest.getSize()), getRandomTool(random, i));
147 				if (i < 5) {
148 					chest.setItem(random.nextInt(chest.getSize()), getRandomArmor(random, i));
149 				}
150 			}
151 
152 			chest.setItem(random.nextInt(chest.getSize()), getRandomOre(random));
153 		} else {
154 			Bukkit.getServer().getLogger().severe("Chest is not giving correct state, returned "
155 					+ state
156 					+ " (block type: "
157 					+ block.getType()
158 					+ ", state type: " + state.getType() + ")");
159 		}
160 	}
