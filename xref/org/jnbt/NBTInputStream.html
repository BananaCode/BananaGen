36  import java.io.Closeable;

38  import java.io.IOException;
39  import java.io.InputStream;

42  import java.util.List;
43  import java.util.Map;
44  import java.util.zip.GZIPInputStream;

46  /**
47   * <p>This class reads <strong>NBT</strong>, or
48   * <strong>Named Binary Tag</strong> streams, and produces an object graph of
49   * subclasses of the <code>Tag</code> object.</p>
50   * 
51   * <p>The NBT format was created by Markus Persson, and the specification may
52   * be found at <a href="http://www.minecraft.net/docs/NBT.txt">
53   * http://www.minecraft.net/docs/NBT.txt</a>.</p>
54   * @author Graham Edgecombe
55   *
56   */
57  public final class NBTInputStream implements Closeable {
58  	
59  	/**
60  	 * The data input stream.
61  	 */
62  	private final DataInputStream is;
63  	
64  	/**
65  	 * Creates a new <code>NBTInputStream</code>, which will source its data
66  	 * from the specified input stream.
67  	 * @param is The input stream.
68  	 * @throws IOException if an I/O error occurs.
69  	 */
70  	public NBTInputStream(InputStream is) throws IOException {
71  		this.is = new DataInputStream(new GZIPInputStream(is));
72  	}
73  	
74  	/**
75  	 * Reads an NBT tag from the stream.
76  	 * @return The tag that was read.
77  	 * @throws IOException if an I/O error occurs.
78  	 */
79  	public Tag readTag() throws IOException {
80  		return readTag(0);
81  	}
82  	
83  	/**
84  	 * Reads an NBT from the stream.
85  	 * @param depth The depth of this tag.
86  	 * @return The tag that was read.
87  	 * @throws IOException if an I/O error occurs.
88  	 */
89  	private Tag readTag(int depth) throws IOException {
90  		int type = is.readByte() & 0xFF;
91  				
92  		String name;
93  		if(type != NBTConstants.TYPE_END) {
94  			int nameLength = is.readShort() & 0xFFFF;
95  			byte[] nameBytes = new byte[nameLength];
96  			is.readFully(nameBytes);
97  			name = new String(nameBytes, NBTConstants.CHARSET);
98  		} else {
99  			name = "";
100 		}
101 		
102 		return readTagPayload(type, name, depth);
103 	}
104 
105 	/**
106 	 * Reads the payload of a tag, given the name and type.
107 	 * @param type The type.
108 	 * @param name The name.
109 	 * @param depth The depth.
110 	 * @return The tag.
111 	 * @throws IOException if an I/O error occurs.
112 	 */
113 	private Tag readTagPayload(int type, String name, int depth) throws IOException {
114 		switch(type) {
115 		case NBTConstants.TYPE_END:
116 			if(depth == 0) {
117 				throw new IOException("TAG_End found without a TAG_Compound/TAG_List tag preceding it.");
118 			} else {
119 				return new EndTag();
120 			}
121 		case NBTConstants.TYPE_BYTE:
122 			return new ByteTag(name, is.readByte());
123 		case NBTConstants.TYPE_SHORT:
124 			return new ShortTag(name, is.readShort());
125 		case NBTConstants.TYPE_INT:
126 			return new IntTag(name, is.readInt());
127 		case NBTConstants.TYPE_LONG:
128 			return new LongTag(name, is.readLong());
129 		case NBTConstants.TYPE_FLOAT:
130 			return new FloatTag(name, is.readFloat());
131 		case NBTConstants.TYPE_DOUBLE:
132 			return new DoubleTag(name, is.readDouble());
133 		case NBTConstants.TYPE_BYTE_ARRAY:
134 			int length = is.readInt();
135 			byte[] bytes = new byte[length];
136 			is.readFully(bytes);
137 			return new ByteArrayTag(name, bytes);
138 		case NBTConstants.TYPE_STRING:
139 			length = is.readShort();
140 			bytes = new byte[length];
141 			is.readFully(bytes);
142 			return new StringTag(name, new String(bytes, NBTConstants.CHARSET));
143 		case NBTConstants.TYPE_LIST:
144 			int childType = is.readByte();
145 			length = is.readInt();
146 			
147 			List<Tag> tagList = new ArrayList<Tag>();
148 			for(int i = 0; i < length; i++) {
149 				Tag tag = readTagPayload(childType, "", depth + 1);
150 				if(tag instanceof EndTag) {
151 					throw new IOException("TAG_End not permitted in a list.");
152 				}
153 				tagList.add(tag);
154 			}
155 			
156 			return new ListTag(name, NBTUtils.getTypeClass(childType), tagList);
157 		case NBTConstants.TYPE_COMPOUND:
158 			Map<String, Tag> tagMap = new HashMap<String, Tag>();
159 			while(true) {
160 				Tag tag = readTag(depth + 1);
161 				if(tag instanceof EndTag) {
162 					break;
163 				} else {
164 					tagMap.put(tag.getName(), tag);
165 				}
166 			}
167 			
168 			return new CompoundTag(name, tagMap);
169 		default:
170 			throw new IOException("Invalid tag type: " + type + ".");
171 		}
172 	}
173 
174 	@Override
175 	public void close() throws IOException {
176 		is.close();
177 	}
